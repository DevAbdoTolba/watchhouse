   ```python
                   # Distinguish "transient decode error" from "cap was released by watchdog"
                   # NOTE: Uses the FrameSource Protocol's `is_open` property (additive
                   # extension defined below) — NOT the private `_cap` attribute. This
                   # keeps the reader decoupled from `_BaseCvCapture` internals, so any
                   # future FrameSource implementation (or test stub) can signal "capture
                   # was released externally" via `is_open` without re-exposing `_cap`.
                   if not self.frame_source.is_open:
                       _LOG.warning(
                           "reader_cap_released_externally camera_id=%s — reconnecting",
                           self.camera_id,
                       )
                       break  # exit inner loop, outer loop will re-open
                   # Otherwise it's a transient decode error; keep trying the same cap
                   # (live RTSP never exits on a single decode error — CONTEXT D-12).
       finally:
           try:
               self.frame_source.release()
           except Exception:
               pass
   ```
